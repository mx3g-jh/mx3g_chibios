83 /**
84  * @brief Configures and activates the WSPI peripheral.
85  *
86  * @param[in] wspip pointer to the @p WSPIDriver object
87  * @param[in] config pointer to the @p WSPIConfig object
88  * @return The operation status.
89  *
90  * @api
91  */
92 msg_t wspiStart(WSPIDriver *wspip, const WSPIConfig *config) {
93  msg_t msg;
94 
95  osalDbgCheck((wspip != NULL) && (config != NULL));
96 
97  osalSysLock();
98  osalDbgAssert((wspip->state == WSPI_STOP) || (wspip->state == WSPI_READY),
99  "invalid state");
100 
101  wspip->config = config;
102 
103 #if defined(WSPI_LLD_ENHANCED_API)
104  msg = wspi_lld_start(wspip);
105 #else
106  wspi_lld_start(wspip);
107  msg = HAL_RET_SUCCESS;
108 #endif
109  if (msg == HAL_RET_SUCCESS) {
110  wspip->state = WSPI_READY;
111  }
112  else {
113  wspip->state = WSPI_STOP;
114  }
115 
116  osalSysUnlock();
117 
118  return msg;
119 }

221 /**
222  * @brief Sends a command without data phase.
223  * @pre In order to use this function the option @p WSPI_USE_WAIT must be
224  * enabled.
225  * @pre In order to use this function the driver must have been configured
226  * without callbacks (@p end_cb = @p NULL).
227  *
228  * @param[in] wspip pointer to the @p WSPIDriver object
229  * @param[in] cmdp pointer to the command descriptor
230  * @return The operation status.
231  * @retval false if the operation succeeded.
232  * @retval true if the operation failed because HW issues.
233  *
234  * @api
235  */
236 bool wspiCommand(WSPIDriver *wspip, const wspi_command_t *cmdp) {
237  msg_t msg;
238 
239  osalDbgCheck((wspip != NULL) && (cmdp != NULL));
240  osalDbgCheck((cmdp->cfg & WSPI_CFG_DATA_MODE_MASK) == WSPI_CFG_DATA_MODE_NONE);
241 
242  osalSysLock();
243 
244  osalDbgAssert(wspip->state == WSPI_READY, "not ready");
245  osalDbgAssert(wspip->config->end_cb == NULL, "has callback");
246 
247  wspiStartCommandI(wspip, cmdp);
248  msg = osalThreadSuspendS(&wspip->thread);
249 
250  osalSysUnlock();
251 
252  return (bool)(msg != MSG_OK);
253 }
