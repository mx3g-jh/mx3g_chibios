87 /**
88  * @brief Configures and activates the MAC peripheral.
89  *
90  * @param[in] macp pointer to the @p MACDriver object
91  * @param[in] config pointer to the @p MACConfig object
92  * @return The operation status.
93  *
94  * @api
95  */
96 msg_t macStart(MACDriver *macp, const MACConfig *config) {
97  msg_t msg;
98 
99  osalDbgCheck((macp != NULL) && (config != NULL));
100 
101  osalSysLock();
102  osalDbgAssert(macp->state == MAC_STOP,
103  "invalid state");
104 
105  macp->config = config;
106 
107 #if defined(MAC_LLD_ENHANCED_API)
108  msg = mac_lld_start(macp);
109 #else
110  mac_lld_start(macp);
111  msg = HAL_RET_SUCCESS;
112 #endif
113  if (msg == HAL_RET_SUCCESS) {
114  macp->state = MAC_ACTIVE;
115  }
116  else {
117  macp->state = MAC_STOP;
118  }
119 
120  osalSysUnlock();
121 
122  return msg;
123 }

148 /**
149  * @brief Allocates a transmission descriptor.
150  * @details One of the available transmission descriptors is locked and
151  * returned. If a descriptor is not currently available then the
152  * invoking thread is queued until one is freed.
153  *
154  * @param[in] macp pointer to the @p MACDriver object
155  * @param[out] tdp pointer to a @p MACTransmitDescriptor structure
156  * @param[in] timeout the number of ticks before the operation timeouts,
157  * the following special values are allowed:
158  * - @a TIME_IMMEDIATE immediate timeout.
159  * - @a TIME_INFINITE no timeout.
160  * .
161  * @return The operation status.
162  * @retval MSG_OK the descriptor was obtained.
163  * @retval MSG_TIMEOUT the operation timed out, descriptor not initialized.
164  *
165  * @api
166  */
167 msg_t macWaitTransmitDescriptor(MACDriver *macp,
168  MACTransmitDescriptor *tdp,
169  sysinterval_t timeout) {
170  msg_t msg;
171 
172  osalDbgCheck((macp != NULL) && (tdp != NULL));
173  osalDbgAssert(macp->state == MAC_ACTIVE, "not active");
174 
175  osalSysLock();
176 
177  while ((msg = mac_lld_get_transmit_descriptor(macp, tdp)) != MSG_OK) {
178  msg = osalThreadEnqueueTimeoutS(&macp->tdqueue, timeout);
179  if (msg == MSG_TIMEOUT) {
180  break;
181  }
182  }
183 
184  osalSysUnlock();
185 
186  return msg;
187 }
