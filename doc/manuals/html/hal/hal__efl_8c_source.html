88 /**
89  * @brief Configures and starts the driver.
90  *
91  * @param[in] eflp pointer to a @p EFlashDriver structure
92  * @param[in] config pointer to a configuration structure.
93  * If this parameter is set to @p NULL then a default
94  * configuration is used.
95  * @return The operation status.
96  *
97  * @api
98  */
99 msg_t eflStart(EFlashDriver *eflp, const EFlashConfig *config) {
100  msg_t msg;
101 
102  osalDbgCheck(eflp != NULL);
103 
104  osalSysLock();
105 
106  osalDbgAssert((eflp->state == FLASH_STOP) || (eflp->state == FLASH_READY),
107  "invalid state");
108 
109  eflp->config = config;
110 
111 #if defined(EFL_LLD_ENHANCED_API)
112  msg = efl_lld_start(eflp);
113 #else
114  efl_lld_start(eflp);
115  msg = HAL_RET_SUCCESS;
116 #endif
117  if (msg == HAL_RET_SUCCESS) {
118  eflp->state = FLASH_READY;
119  }
120  else {
121  eflp->state = FLASH_STOP;
122  }
123 
124  osalSysUnlock();
125 
126  return msg;
127 }
