74 /**
75  * @brief Configures and activates the TRNG peripheral.
76  *
77  * @param[in] trngp pointer to the @p TRNGDriver object
78  * @param[in] config pointer to the @p TRNGConfig object or @p NULL for
79  * default configuration
80  * @return The operation status.
81  *
82  * @api
83  */
84 msg_t trngStart(TRNGDriver *trngp, const TRNGConfig *config) {
85  msg_t msg;
86 
87  osalDbgCheck(trngp != NULL);
88 
89  osalSysLock();
90  osalDbgAssert((trngp->state == TRNG_STOP) || (trngp->state == TRNG_READY),
91  "invalid state");
92 
93  trngp->config = config;
94 
95 #if defined(TRNG_LLD_ENHANCED_API)
96  msg = trng_lld_start(trngp);
97 #else
98  trng_lld_start(trngp);
99  msg = HAL_RET_SUCCESS;
100 #endif
101  if (msg == HAL_RET_SUCCESS) {
102  trngp->state = TRNG_READY;
103  }
104  else {
105  trngp->state = TRNG_STOP;
106  }
107 
108  osalSysUnlock();
109 
110  return msg;
111 }

113 /**
114  * @brief Deactivates the TRNG peripheral.
115  *
116  * @param[in] trngp pointer to the @p TRNGDriver object
117  *
118  * @api
119  */
120 void trngStop(TRNGDriver *trngp) {
121 
122  osalDbgCheck(trngp != NULL);
123 
124  osalSysLock();
125 
126  osalDbgAssert((trngp->state == TRNG_STOP) || (trngp->state == TRNG_READY),
127  "invalid state");
128 
129  trng_lld_stop(trngp);
130  trngp->config = NULL;
131  trngp->state = TRNG_STOP;
132 
133  osalSysUnlock();
134 }

136 /**
137  * @brief True random numbers generator.
138  * @note The function is blocking and likely performs polled waiting
139  * inside the low level implementation.
140  *
141  * @param[in] trngp pointer to the @p TRNGDriver object
142  * @param[in] size size of output buffer
143  * @param[out] out output buffer
144  * @return The operation status.
145  * @retval false if a random number has been generated.
146  * @retval true if an HW error occurred.
147  *
148  * @api
149  */
150 bool trngGenerate(TRNGDriver *trngp, size_t size, uint8_t *out) {
151  bool err;
152 
153  osalDbgCheck((trngp != NULL) && (out != NULL));
154 
155  osalDbgAssert(trngp->state == TRNG_READY, "not ready");
156 
157  trngp->state = TRNG_RUNNING;
158 
159  err = trng_lld_generate(trngp, size, out);
160 
161  trngp->state = TRNG_READY;
162 
163  return err;
164 }
