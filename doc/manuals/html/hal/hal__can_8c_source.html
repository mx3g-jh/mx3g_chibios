92 /**
93  * @brief Configures and activates the CAN peripheral.
94  * @note Activating the CAN bus can be a slow operation.
95  * @note Unlike other drivers it is not possible to restart the CAN
96  * driver without first stopping it using canStop().
97  *
98  * @param[in] canp pointer to the @p CANDriver object
99  * @param[in] config pointer to the @p CANConfig object. Depending on
100  * the implementation the value can be @p NULL.
101  * @return The operation status.
102  *
103  * @api
104  */
105 msg_t canStart(CANDriver *canp, const CANConfig *config) {
106  msg_t msg;
107 
108  osalDbgCheck(canp != NULL);
109 
110  osalSysLock();
111  osalDbgAssert(canp->state == CAN_STOP, "invalid state");
112 
113  /* Entering initialization mode. */
114  canp->state = CAN_STARTING;
115  canp->config = config;
116 
117  /* Low level initialization, could be a slow process and sleeps could
118  be performed inside.*/
119 #if defined(CAN_LLD_ENHANCED_API)
120  msg = can_lld_start(canp);
121 #else
122  can_lld_start(canp);
123  msg = HAL_RET_SUCCESS;
124 #endif
125  if (msg == HAL_RET_SUCCESS) {
126  canp->state = CAN_READY;
127  }
128 
129  osalSysUnlock();
130 
131  return msg;
132 }

134 /**
135  * @brief Deactivates the CAN peripheral.
136  *
137  * @param[in] canp pointer to the @p CANDriver object
138  *
139  * @api
140  */
141 void canStop(CANDriver *canp) {
142 
143  osalDbgCheck(canp != NULL);
144 
145  osalSysLock();
146  osalDbgAssert((canp->state == CAN_STOP) || (canp->state == CAN_READY),
147  "invalid state");
148 
149  /* The low level driver is stopped.*/
150  canp->state = CAN_STOPPING;
151  can_lld_stop(canp);
152  canp->config = NULL;
153  canp->state = CAN_STOP;
154 
155  /* Threads waiting on CAN APIs are notified that the driver has been
156  stopped in order to not have stuck threads.*/
157  osalThreadDequeueAllI(&canp->rxqueue, MSG_RESET);
158  osalThreadDequeueAllI(&canp->txqueue, MSG_RESET);
159  osalOsRescheduleS();
160  osalSysUnlock();
161 }

163 /**
164  * @brief Can frame transmission attempt.
165  * @details The specified frame is queued for transmission, if the hardware
166  * queue is full then the function fails.
167  *
168  * @param[in] canp pointer to the @p CANDriver object
169  * @param[in] mailbox mailbox number, @p CAN_ANY_MAILBOX for any mailbox
170  * @param[in] ctfp pointer to the CAN frame to be transmitted
171  * @return The operation result.
172  * @retval false Frame transmitted.
173  * @retval true Mailbox full.
174  *
175  * @iclass
176  */
177 bool canTryTransmitI(CANDriver *canp,
178  canmbx_t mailbox,
179  const CANTxFrame *ctfp) {
180 
181  osalDbgCheckClassI();
182  osalDbgCheck((canp != NULL) && (ctfp != NULL) &&
183  (mailbox <= (canmbx_t)CAN_TX_MAILBOXES));
184  osalDbgAssert((canp->state == CAN_READY) || (canp->state == CAN_SLEEP),
185  "invalid state");
186 
187  /* If the RX mailbox is full then the function fails.*/
188  if (!can_lld_is_tx_empty(canp, mailbox)) {
189  return true;
190  }
191 
192  /* Transmitting frame.*/
193  can_lld_transmit(canp, mailbox, ctfp);
194 
195  return false;
196 }

250 /**
251  * @brief Can frame transmission.
252  * @details The specified frame is queued for transmission, if the hardware
253  * queue is full then the invoking thread is queued.
254  * @note Trying to transmit while in sleep mode simply enqueues the thread.
255  *
256  * @param[in] canp pointer to the @p CANDriver object
257  * @param[in] mailbox mailbox number, @p CAN_ANY_MAILBOX for any mailbox
258  * @param[in] ctfp pointer to the CAN frame to be transmitted
259  * @param[in] timeout the number of ticks before the operation timeouts,
260  * the following special values are allowed:
261  * - @a TIME_IMMEDIATE immediate timeout.
262  * - @a TIME_INFINITE no timeout.
263  * .
264  * @return The operation result.
265  * @retval MSG_OK the frame has been queued for transmission.
266  * @retval MSG_TIMEOUT The operation has timed out.
267  * @retval MSG_RESET The driver has been stopped while waiting.
268  *
269  * @api
270  */
271 msg_t canTransmitTimeout(CANDriver *canp,
272  canmbx_t mailbox,
273  const CANTxFrame *ctfp,
274  sysinterval_t timeout) {
275 
276  osalDbgCheck((canp != NULL) && (ctfp != NULL) &&
277  (mailbox <= (canmbx_t)CAN_TX_MAILBOXES));
278 
279  osalSysLock();
280  osalDbgAssert((canp->state == CAN_READY) || (canp->state == CAN_SLEEP),
281  "invalid state");
282 
283  /*lint -save -e9007 [13.5] Right side is supposed to be pure.*/
284  while ((canp->state == CAN_SLEEP) || !can_lld_is_tx_empty(canp, mailbox)) {
285  /*lint -restore*/
286  msg_t msg = osalThreadEnqueueTimeoutS(&canp->txqueue, timeout);
287  if (msg != MSG_OK) {
288  osalSysUnlock();
289  return msg;
290  }
291  }
292  can_lld_transmit(canp, mailbox, ctfp);
293  osalSysUnlock();
294  return MSG_OK;
295 }
